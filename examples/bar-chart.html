const data = [
  { product: 'Product A', revenue: 420 },
  { product: 'Product B', revenue: 580 },
  { product: 'Product C', revenue: 310 },
  { product: 'Product D', revenue: 620 },
  { product: 'Product E', revenue: 400 }
];

const spec = {
  type: 'barChart',
  data: data,
  x: { field: 'product' },
  y: { field: 'revenue' },
  color: '#3366CC',
  title: 'Product Revenue'
};

Devize.createBarChart(spec, document.getElementById('chart-container'));
