Group Visualization

This example demonstrates grouping multiple shapes together with Devize.

Code

const group = buildViz({
  type: "group",
  transform: "translate(50, 50)",
  children: [
    {
      type: "rectangle",
      x: 0,
      y: 0,
      width: 100,
      height: 60,
      fill: "#6699CC",
      stroke: "#336699",
      strokeWidth: 2,
      rx: 5,
      ry: 5
    },
    {
      type: "circle",
      cx: 150,
      cy: 30,
      r: 25,
      fill: "#CC6666",
      stroke: "#994444",
      strokeWidth: 2
    },
    {
      type: "line",
      x1: 0,
      y1: 100,
      x2: 200,
      y2: 100,
      stroke: "#666666",
      strokeWidth: 2
    }
  ]
}, document.getElementById("visualization-container"));
