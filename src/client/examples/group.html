Group Visualization

This example demonstrates creating a group of shapes with Devize.

Code

const shapesViz = buildViz({
  type: "group",
  children: [
    {
      type: "rectangle",
      x: 50,
      y: 50,
      width: 100,
      height: 80,
      fill: "#6699CC",
      stroke: "#336699",
      strokeWidth: 2,
      rx: 5,
      ry: 5
    },
    {
      type: "circle",
      cx: 250,
      cy: 80,
      r: 40,
      fill: "#CC6666",
      stroke: "#994444",
      strokeWidth: 2
    },
    {
      type: "line",
      x1: 50,
      y1: 200,
      x2: 350,
      y2: 200,
      stroke: "#666666",
      strokeWidth: 3,
      strokeDasharray: "5,5"
    }
  ]
}, document.getElementById("visualization-container"));
